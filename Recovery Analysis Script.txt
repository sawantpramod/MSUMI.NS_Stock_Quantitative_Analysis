# time spend in drawdown
# Time in Drawdown	Meaning
# < 30%	Strong stock
# 30-50%	Healthy
# 50-70%	Weak
# >70%	Persistent underperformance

drawdown_days = (df['drawdown'] < 0).sum()

total_days = len(df)

drawdown_percentage = (
    drawdown_days /
    total_days
) * 100

print(f"\nTime in Drawdown: {drawdown_percentage:.2f}%")
